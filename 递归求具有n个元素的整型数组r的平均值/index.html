int get_agv(int R[], int n)
{
	if(n==1)
		return R[0];
	else
		return (R[n-1] + get_agv(R, n-1)*(n-1))/n;	   // n-1  用来约分母(约前面一个数的分母)， 把分母写在横线的下面比较容易看出来
}


void main()
{
	int R[5] = {22,44,66,88,10};
	printf("平均值: %d\n", get_agv(R, 5));
}
